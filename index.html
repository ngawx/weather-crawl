  <style>

    body {

      font-family: Arial, sans-serif;

      margin: 0;

      padding: 0;

      overflow: hidden;

      background-color: #000;

    }

    .alert-crawl {

      position: fixed;

      bottom: 0;

      width: 100%;

      background: linear-gradient(90deg, #ff0000, #ff9900);

      color: #fff;

      font-size: 1.5rem;

      white-space: nowrap;

      overflow: hidden;

      box-shadow: 0px -3px 10px rgba(0, 0, 0, 0.5);

    }

    .alert-text {

      display: inline-block;

      padding: 10px 0;

      animation: scroll 20s linear infinite;

    }

    @keyframes scroll {

      0% { transform: translateX(100%); }

      100% { transform: translateX(-100%); }

    }

    @media (max-width: 600px) {

      .alert-crawl {

        font-size: 1rem;

<body>

  <div class="alert-crawl">

    <div id="alert-text" class="alert-text">Fetching alerts...</div>

  </div>


  <script>

    const alertText = document.getElementById("alert-text");


    async function updateAlerts() {

      try {

        // Simulated fetch from a server-side alerts.json file

        const response = await fetch("alerts.json");

        const alerts = await response.json();

        alertText.innerHTML = alerts.length ? alerts.join(" | ") : "No active alerts.";

        setScrollSpeed();

      } catch (error) {

        alertText.innerHTML = "Error fetching alerts. Please try again later.";

      }

    }


    function setScrollSpeed() {

      const textWidth = alertText.scrollWidth;

      const crawlWidth = document.querySelector(".alert-crawl").offsetWidth;

      const animationDuration = (textWidth + crawlWidth) / 100; // Adjust speed factor

      alertText.style.animationDuration = `${animationDuration}s`;

    }


    // Simulate fetching updates every 60 seconds

    setInterval(updateAlerts, 60000);

    updateAlerts();

  </script>

</body>
